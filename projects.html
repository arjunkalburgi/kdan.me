class Field(object):

def init(self):
"""
Creates and builds a randomly generated graph that the field lies on.
The graph has vertices that are coordinates on a grid
Generates walls randomly with a percentage approximately equal to the wall density
The graph will always be completely connected (i.e. never greater than 1 subgraph)
and each vertex will have at least 2 neighbours, i.e. and entry point and exit point.
"""
self._started = True
wall_prob = self._wall_density
while True:
for x in range(0, self._width):
self._grid.insert(x, [])
wall_set = set()
valid = True
g = Graph()
# loop through each tile and decide if it should be a wall or not
for i in range(0, self.height):
for j in range(0, self.width):
prob = randint(0, 100)
# if its a wall, make it a wall
prob < wall_prob:
wall_set.add((j, i))
self._grid[j].insert(i, grid(j, i, True))
else:
g.add_vertex((j, i))
self._grid[j].insert(i, grid(j, i, False))
# if its not the top row, add up edge
if i > 0 and (j, i-1) not in wall_set:
g.add_edge(((j, i-1), (j, i)))
g.add_edge(((j, i), (j, i-1)))
# if its not the left-most column, add left edge
if j > 0 and (j-1, i) not in wall_set:
g.add_edge(((j-1, i), (j, i)))
g.add_edge(((j, i), (j-1, i)))

# makes sure the graph has only 1 subgraph
components = count_components(g)
if components != 1:
continue
# makes sure each passable tile has at least 2 neighbours for entry and exit
for v in g.vertices():
if len(g.neighbours(v)) < 2:
valid = False
break
# print("Subgraphs = " + str(components))
# if 1 subgraph and each passable tile has at least 2 neighbours, the graph
# is a valid map and should be generated. Otherwise make a new one
if components == 1 and valid:
break
else:
self._grid = []
